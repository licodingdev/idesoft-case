cURL Examples

        
        
        

            

                GET
                /orders
            

            

                
curl -X GET http://localhost:8080/orders

            

        


        
        

            

                POST
                /orders
            

            

                
curl -X POST http://localhost:8080/orders \
  -H "Content-Type: application/json" \
  -d '{
    "customerId": 1,
    "items": [
      {
        "productId": 100,
        "quantity": 10,
        "unitPrice": 120.75
      }
    ]
  }'

            

        


        
        

            

                DELETE
                /orders/{id}
            

            

                
curl -X DELETE http://localhost:8080/orders/1

            

        


        
        

            

                GET
                /orders/{id}/discounts
            

            

                
curl -X GET http://localhost:8080/orders/1/discounts
